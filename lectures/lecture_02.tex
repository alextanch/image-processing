\frametitle{Взвешенная сумма изображений}
\begin{itemize}
    \item изображения одинакового размера $H\times W$
    $$
    A = \left\{a_{i,j}\right\}\,,\quad B = \left\{b_{i,j}\right\}\,,\quad
    i=1,2,\ldots,H\,,\quad
    j=1,2,\ldots,W
    $$
    \item взвешенная сумма
    $$
        C=\alpha\cdot A + \beta\cdot B
    $$
    \item сумма изображений при $\alpha=1$, $\beta=1$
    $$
        C=A+B
    $$
    \item разность изображений при $\alpha=1$, $\beta=-1$
    $$
        C=A-B
    $$
    \item blending при $0<\alpha<1$ и $\beta=1-\alpha$
    $$
        C=\alpha\cdot A+(1-\alpha)\cdot B
    $$
\end{itemize}
\end{frame}

\begin{frame}
    \frametitle{Уменьшение шума при сложении изображений}
    Изображения с аддитивным гауссовским шумом
    $$
    A_n = A + \varepsilon_n\,,\quad \varepsilon_n\sim\mathcal{N}(0,\sigma^2)\,,\quad
    n=1,2\ldots,N
    $$
    Усредним зашумленные изображения
    $$
        B=\frac{1}{N}\sum_{n=1}^N A_n
    $$
    Математическое ожидание
    $$
        \mathbf{E}[B]=\frac{1}{N}\sum_{n=1}^N\left(\mathbf{E}[A]+\mathbf{E}[\varepsilon_n]\right)=A
    $$
    Дисперсия
    $$
        \mathbf{D}[B]=\frac{1}{N^2}\sum_{n=1}^N\left(\mathbf{D}[A]+\mathbf{D}[\varepsilon_n]\right)=
        \frac{\sigma^2}{N}
    $$
\end{frame}

\begin{frame}
    \frametitle{Гистограмма изображения}
    $A$ -- одноканальное, $b$ - битное изображение размера $H\times W$
    $$
        A=\left\{a_{ij}\right\},\quad
        a_{ij}\in\left\{0,1,\ldots,L-1\right\},\quad
        L=2^b,\quad
        i=1,2,\ldots ,H,\quad
        j=1,2,\ldots ,W
    $$
    \begin{itemize}
        \item гистограмма изображения
        $$
            n_l = \left|\left\{a_{ij}: a_{ij}=l\right\}\right|\,,\quad
            l=0,1,\ldots ,L-1
        $$
        \item $p_l$ -- вероятность появления пикселя со значением яркости $l$
        $$
            p_l=\frac{n_l}{H\cdot W},\qquad
            l=0,1,\ldots ,L-1
        $$
         \item сумма вероятностей равна единице
        $$
            \sum_{l=0}^{L-1}p_l=\frac{1}{H\cdot W}\sum_{l=0}^{L-1}n_l=1
        $$
    \end{itemize}
\end{frame}

\begin{frame}
    \frametitle{Преобразование случайной величины к равномерному распределению}
    \begin{itemize}
        \item $X \sim p(x)$ -- непрерывная cлучайная величина $X$ имеет плотность $p(x)$
        \item функция распределения случайной величины
        $$
            y=F(x)=\mathbf{P}[X<x]=\int\limits_0^xp(t)\,dt
        $$
        \item новая случайная величина $Y=F\left(X\right)$ будет иметь распределение
        $$
            q(y)=p(x)\left|x'_y\right|
        $$
        \item вычислим $q(y)$
        $$
            y'_x=\frac{d}{dx}\int\limits_0^xp(t)\,dt=p(x)
            \quad\Rightarrow\quad 
            x'_y=\frac{1}{p(x)}
            \quad\Rightarrow\quad
            q(y)=p(x)\left|\frac{1}{p(x)}\right|=1
        $$
        \item таким образом случайная величина $Y$ распределена равномерно
    \end{itemize}
\end{frame}

%---------------------------------------------------------------------
\begin{frame}
    \frametitle{Эквализации гистограммы изображения}
    \begin{itemize}
        \item непрерывная равномерная случайная величина
        $$
            y=F(x)=\mathbf{P}[X<x]=\int\limits_0^xp(t)\,dt
        $$
        \item если преобразовать значения пикселей по формуле
        $$
            y_l=\sum_{m=0}^l p_m\,,\quad y_l\in[0,1]
        $$
        \item то получим примерно равномерное распределение
    \end{itemize}
\end{frame}

\begin{frame}
\frametitle{Применение интегральных изображений для вычисления локальных статистик}
    \begin{figure}
            \centering
            \includegraphics[height=0.3\textheight]{lochist1.pdf}
    \end{figure}
    \begin{itemize}
        \item выборочное среднее в прямоугольнике $S_{xy}$
        $$
            \mu_{xy}=\frac{1}{H\cdot W}\sum_{i,j\in S_{xy}} a(i,j)
        $$
        \item выборочная дисперсия
        $$
            \sigma^2_{xy}=
            \frac{1}{H\cdot W}\sum_{i,j\in S_{xy}} \left[a(i,j)-\mu\right]^2=
            \frac{1}{H\cdot W}\sum_{i,j\in S_{xy}} a^2(i,j) - \mu^2(x,y)
        $$
    \end{itemize}
\end{frame}

\begin{frame}
    \frametitle{Поворот и масштабирование изображения}
    \begin{figure}[t]
        \includegraphics[height=0.33\textheight]{rotation.png}
    \end{figure}
    \begin{itemize}
        \item прямое преобразование $(x,y)\mapsto(x',y')$
        $$
            \begin{bmatrix}
                x' \\ y'
            \end{bmatrix} =
            s\cdot\mathbf{R}
            \begin{bmatrix} 
                x - c_x \\ y - c_y
            \end{bmatrix} +
            \begin{bmatrix} 
                c_x \\ c_y
            \end{bmatrix}=
            s\cdot\mathbf{R}
            \begin{bmatrix}
                x \\ y
            \end{bmatrix}+
            \mathbf{t}
        $$
        $$
            \mathbf{R}=
            \begin{bmatrix*}[r]
                \cos\theta & -\sin\theta \\
                \sin\theta & \cos\theta
            \end{bmatrix*}\,,\quad
            \mathbf{t}=
            \left(\mathbf{I}-s\cdot\mathbf{R}\right)
            \begin{bmatrix} 
                c_x \\ c_y
            \end{bmatrix}
        $$
        \item обратное преобразование $(x',y')\mapsto(x,y)$
        $$
            \begin{bmatrix} 
                x \\ y
            \end{bmatrix}=
            \frac{1}{s}\mathbf{R}^T
            \left(
            \begin{bmatrix} 
                x' \\ y'
            \end{bmatrix}-
            \mathbf{t}
            \right)
        $$
    \end{itemize}
\end{frame}

            g(x', y')=f(x_i,y_i)
        $$
    \end{itemize}

\begin{frame}
    \frametitle{Билинейная интерполяция изображения}
    \begin{figure}[t]
        \includegraphics[height=0.4\textheight]{bilinear.png}
    \end{figure}
    $$
        g(x', y')=a_1+a_2\cdot x + a_3\cdot y + a_4\cdot x\cdot y
    $$
    Коэффициенты находятся из условий
    $$
        a_1+a_2\cdot x_i + a_3\cdot y_i + a_4\cdot x_i\cdot y_i=f(x_i,y_i)\,,\quad
        i=1,2,3,4
    $$
    В итоге получим
    $$
        g(x', y')=f(x_1,y_2)(1-x)(1-y)+f(x_1,y_1)x(1-y)+f(x_2,y_2)(1-x)y+f(x_2,y_1)xy
    $$
\end{frame}
