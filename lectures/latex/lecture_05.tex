\begin{frame}
\frametitle{Итеративный алгоритм подбора порога}
\begin{itemize}
    \item[1.]<1-> выбирать некоторую начальную оценку значения порога $T$
    \item[2.]<2-> сегментировать изображение с помощью порога $T$
    $$
        g(x, y) =
        \begin{cases}
            1\,, & f(x,y) > T\\
            0\,, & f(x,y) \leq T
        \end{cases}
    $$
    в результате образуются два класса пикселей: $C_1$ и $C_2$.
    \item[3.]<3-> вычислить значения $\mu(C_1)$ и $\mu(C_2)$ средних значений
  яркостей пикселей в каждом классе $C_1$ и $C_2$
    \item[4.]<4-> вычислить новое значение порога
    $$
        T = \frac{\mu(C_1)+\mu(C_2)}{2}
    $$
    \item[5.]<5-> повторять шаги 2-4 до тех пор, пока разница значений $T$
    при соседних итерациях не окажется меньше значения наперед заданного параметра $\Delta T$
\end{itemize}
\end{frame}

%------------------------------------------------------------------------
\begin{frame}
\frametitle{Метод Оцу}

Подберем порог таким образом, что бы дисперсия в каждом классе была минимальна.

\medskip

\textbf{Предварительные вычисления.}

\begin{itemize}
    \item для изображения $f(x,y)$ размера $H\times W$ вычислим гистограмму
    $$
        \left\{n_{\,i}\right\}\,,\quad i=0,1,2,\ldots,255
    $$
    \item для каждого значения $i=0,1,2,\ldots,255$ вычислим вероятность того,
    что пиксель принимает значение $i$ и функцию распределения
    $$
        p_{i}=\mathbf{P}\left[f = i\right]=\frac{n_{i}}{M\cdot N}
    $$
    $$
        F_i=\mathbf{P}\left[f \leqslant i\right]=\sum_{j=0}^i p_{j}
    $$
\end{itemize}
\end{frame}

%------------------------------------------------------------------------
\begin{frame}
\frametitle{Метод Оцу}

\textbf{Вычисление порога методом Оцу:} 
\medskip

для каждого порога $t=1,2,\ldots,255$
\medskip

\begin{itemize}
    \item разобьем все пиксели на два класса
    $$
        C_1 = \left\{(x,y): f(x,y) < t\right\}\,,\quad
        C_2 = \left\{(x,y): f(x,y) \geqslant t\right\}
    $$
    \item вычислим вероятности принадлежности к классам $C_1$ и $C_2$
    $$
        q_1(t) = \sum_{i=0}^t p_{i}=F(t)\,,\quad
    $$
    $$
        q_2(t) =1-q_1(t)
    $$
\end{itemize}
\end{frame}

%------------------------------------------------------------------------
\begin{frame}
\frametitle{Метод Оцу}
\begin{itemize}
    \item найдем математические ожидания значений пикселей в каждом классе
    $$
        \mu_1(t)=\frac{1}{q_1}\sum_{i=0}^t i \cdot p_{i}
    $$
    $$
        \mu_2(t)=\frac{1}{q_2}\sum_{i=t+1}^{255} i \cdot p_{i}
    $$
\end{itemize}
\end{frame}

%------------------------------------------------------------------------
\begin{frame}
\frametitle{Метод Оцу}
\begin{itemize}
    \item вычислим дисперсии значений пикселей в каждом классе
    $$
        \sigma_1^2(t)=\frac{1}{q_1}\sum_{i=0}^t [i-\mu_1(t)]^2 \cdot p_{i}
    $$
    $$
        \sigma_2^2(t)=\frac{1}{q_2}\sum_{i=t+1}^{255} [i-\mu_2(t)]^2 \cdot p_{i}
    $$
\end{itemize}
\end{frame}

%------------------------------------------------------------------------
\begin{frame}
\frametitle{Метод Оцу}
Рассмотрим функцию потерь
$$
    \mbox{Loss}(t)=q_1(t)\sigma_1^2(t)  + q_2(t)\sigma_2^2(t)
$$
Порог Оцу равен
   $$
    T=\arg\min_{0 < t < 256} \mbox{Loss}(t)
   $$
\end{frame}

\begin{frame}
\frametitle{Сегментация с помощью метода скользящего среднего}
\begin{itemize}
    \item обойти изображение $f(x,y)$ размера $H\times W$ в z-порядке по строкам
    $$
        f(x,y) \longmapsto z(k)\,,\quad k=1,2,\ldots,HW
    $$
    \item вычислить скользящее среднее по окну размера $n$
    $$
        m(k)=\frac{1}{n}\sum_{i=k+2-n}^{k+1}z_i=
        m(k) + \frac{z(k+1)-z(k-n+1)}{n}\,,\quad
        m(1)=\frac{z(1)}{n}
    $$
    \item сегментировать изображение
    $$
    g(x, y) =
        \begin{cases}
            1\,, & f(x,y) > a \cdot m(k)\\
            0\,, & \mbox{в противном случае}
        \end{cases}
    $$
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Бинарная морфология изображений}

Бинарная \textbf{морфология} преобразует маски в маски.
\medskip

Основные морфологические операции.
\begin{itemize}
    \item отражение / сдвиг
    \item  эрозия / дилатация
    \item размыкание / замыкание
\end{itemize}
\end{frame}

%------------------------------------------------------------------------
\begin{frame}
\frametitle{Отражение и сдвиг}
\begin{figure}
    \centering
    \includegraphics[width=0.4\textwidth]{fig15.pdf}
\end{figure}
\begin{itemize}
    \item центральное отражение (поворот на $180^{\circ}$) множества
    $$
        \hat{B}=\left\{b: -b\in B\right\}
    $$
    \item сдвиг множества на вектор $z=(z_1,z_2)$
    $$
        B_z=\left\{b: b-z\in B\right\}
    $$
\end{itemize}
\end{frame}

%------------------------------------------------------------------------
\begin{frame}
\frametitle{Эрозия}
\begin{figure}
    \centering
    \includegraphics[width=0.4\textwidth]{fig16.pdf}
\end{figure}
Эрозия множества $A$ по множеству (структурному элементу, ядру) $B$
$$
    A \ominus B = \left\{z: B_z\subset A\right\}=
    \left\{z: B_z\cap A^c=\varnothing\right\}
$$
\end{frame}

%------------------------------------------------------------------------
\begin{frame}
\frametitle{Дилатация}
\begin{figure}
    \centering
    \includegraphics[width=0.45\textwidth]{fig17.pdf}
\end{figure}
$$
    A \oplus B = \left\{z: \hat{B}_z\cap A\ne\varnothing\right\}
$$
\end{frame}

%------------------------------------------------------------------------
\begin{frame}
\frametitle{Двойственность эрозии и дилатации}
$$
    \left(A \ominus B\right)^c = A^c\oplus\hat{B}
$$
\medskip
$$
    \left(A \oplus B\right)^c = A^c\ominus\hat{B}
$$
\end{frame}

%------------------------------------------------------------------------
\begin{frame}
\frametitle{Размыкание (opening)}
\begin{figure}
    \centering
    \includegraphics[width=0.5\textwidth]{fig18.pdf}
\end{figure}
$$
    A \circ B = \left(A\ominus B\right)\oplus B
$$
\end{frame}

%------------------------------------------------------------------------
\begin{frame}
\frametitle{Замыкание (closing)}
\begin{figure}
    \centering
    \includegraphics[width=0.5\textwidth]{fig19.pdf}
\end{figure}
$$
    A \bullet B = \left(A\oplus B\right)\ominus B
$$
\end{frame}

%------------------------------------------------------------------------
\begin{frame}
\frametitle{Свойства размыкания и замыкания}
\begin{itemize}
    \item<1->
    $$
        A\circ B\subset A\,,\quad
        A\subset A\bullet B
    $$
    \item<2->
    $$
        \left(A\circ B\right)\circ B=A\circ B\,,\quad
        \left(A\bullet B\right)\bullet B=A\bullet B
    $$
    \item<3->
    $$
        \left(A\bullet B\right)^c=A^c\circ \hat{B}\,,\quad
        \left(A\circ B\right)^c=A^c\bullet \hat{B}
    $$
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Утолщение}
$$
    A\odot B=A\cup(A\circledast B)
$$
$$
    A\odot\{B^1,B^2,\ldots,B^n\}=
    \left(\left(\cdots\left(\left(A\odot B^1\right)\odot B^2\right)\cdots\right)\odot B^n\right)
$$
Для утолщения множества $A$
сначала строится множество $C = A^c$, проводится его утончение, а затем строится
$C^c$.
