\begin{frame}
\frametitle{Дискретный базис Фурье}
\begin{itemize}
    \item базис Фурье для изображения размера $M\times N$
    $$
        \varphi(\boldsymbol{\omega},\boldsymbol{x}) = e^{-2\pi i(ux/M+vy/N)}\,,\qquad
        \boldsymbol{\omega}=(u,v)\,,\quad
        \mathbf{x}=(x,y)
    $$
    \item $\varphi(\boldsymbol{0},\boldsymbol{x})=1$
    \item $\boldsymbol{\omega}$ -- направление волн, $\norm{\boldsymbol{\omega}}$ --  частота волн
    $$
        \mathrm{Re}\left[\varphi(\boldsymbol{\omega},\boldsymbol{x})\right]=\cos\left[2\pi(ux/M+vy/N)\right]
    $$
    \begin{figure}
        \centering
        \includegraphics[height=0.4\textheight]{fig1.png}
        \caption*{$\boldsymbol{\omega}=(1,2)$\hspace{2cm} $\boldsymbol{\omega}=(5,-5)$}
    \end{figure}
\end{itemize}
\end{frame}

%---------------------------------------------------------------------
\begin{frame}
\frametitle{Дискретное преобразование Фурье (DFT)}
\begin{figure}
    \centering
    \includegraphics[height=0.32\textheight]{fig2.png}
\end{figure}
\begin{itemize}
    \item дискретное преобразование Фурье изображения $f(\boldsymbol{x})$ размера $M\times N$
    $$
        F(\boldsymbol{\omega})=\sum_{x=0}^{M-1}\sum_{y=0}^{N-1}f(\boldsymbol{x})\varphi(\boldsymbol{\omega},\boldsymbol{x})=
        \sum_{\boldsymbol{x}}f(\boldsymbol{x})\varphi(\boldsymbol{\omega},\boldsymbol{x})
    $$
    \item спектр Фурье
    $$
        |F(\boldsymbol{\omega})|
    $$
    \item фазовый спектр Фурье
    $$
        \arg F(\boldsymbol{\omega})
    $$
\end{itemize}
\end{frame}

%---------------------------------------------------------------------
\begin{frame}
\frametitle{Быстрое преобразование Фурье (FFT)}
$$
    F(\boldsymbol{\omega})=\sum_{x=0}^{M-1}\sum_{y=0}^{N-1}f(\boldsymbol{x})\varphi(\boldsymbol{\omega},\boldsymbol{x})
$$    
$$
    \boldsymbol{\omega}=(u,v)\,,\quad
    0\leqslant u < M\,,\quad 0\leqslant v < N
$$
\begin{itemize}
    \item вычислительна сложность (число сложений и умножений) DFT
    $$
        \mathcal{O}[(MN)^2]
    $$ 
    \item вычислительна сложность FFT
    $$
        MN\cdot \mathcal{O}[\ln(MN)]
    $$
\end{itemize}
\end{frame}

\frametitle{Периодичность. Сдвиг спектра}
\begin{figure}
    \centering
    \includegraphics[height=0.3\textheight]{fig4.png}
    \includegraphics[height=0.4\textheight]{fig3.pdf}
    \includegraphics[height=0.3\textheight]{fig5.png}
\end{figure}
\begin{itemize}
    \item значение DFT в самой низкой частоте $\boldsymbol{\omega}=(0,0)$
    $$
        F(\boldsymbol{0})=\sum_{\boldsymbol{x}}f(\boldsymbol{x})\varphi(\boldsymbol{0},\boldsymbol{x})=
        \sum_{\boldsymbol{x}}f(\boldsymbol{x})
    $$ 
    \item периодичность DFT
    $$
        F(u+M,v+N)=F(u,v)
    $$
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Инвариантность спектра Фурье при сдвиге изображения}
\begin{figure}
    \centering
    \includegraphics[height=0.45\textheight]{fig6.png}
\end{figure}
\begin{itemize}
    \item при сдвиге изображения спектр Фурье не меняется
    \begin{align*}
        \left|\sum_{\boldsymbol{x}}f(\boldsymbol{x}+\boldsymbol{x}_0)\varphi(\boldsymbol{\omega},\boldsymbol{x})\right|&=
        \left|e^{2\pi i(x_0u/M+y_0v/N)}\sum_{\boldsymbol{x}'}f(\boldsymbol{x}')\varphi(\boldsymbol{\omega},\boldsymbol{x}')\right|=\\
        &=\left|\sum_{\boldsymbol{x}'}f(\boldsymbol{x}')\varphi(\boldsymbol{\omega},\boldsymbol{x}')\right|
    \end{align*}
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Обратное дискретное преобразование Фурье (IDFT)}
\begin{itemize}
    \item дискретное преобразование Фурье
    $$
        F(\boldsymbol{\omega})=\sum_{\boldsymbol{x}}f(\boldsymbol{x})\varphi(\boldsymbol{\omega},\boldsymbol{x})=
        \sum_{x=0}^{M-1}\sum_{y=0}^{N-1}f(x,y)e^{-2\pi i(ux/M+vy/N)}
    $$
    \item обратное дискретное преобразование Фурье
    $$
        f(\boldsymbol{x})=\frac{1}{MN}\sum_{\boldsymbol{\omega}}F(\boldsymbol{\omega})\varphi^\ast(\boldsymbol{\omega},\boldsymbol{x})
        =\frac{1}{MN}\sum_{u=0}^{M-1}\sum_{v=0}^{N-1}F(u,v)e^{2\pi i(ux/M+vy/N)}
    $$
    \item обозначения
    $$
        F = \mathcal{F}[f]\,,\qquad
        f = \mathcal{F}^{-1}[F]
    $$
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Преобразование Фурье свертки}
\begin{itemize}
    \item  свертка изображения $f$ с ядром $h$
    $$
        (f\circ h)(x,y)=\sum_{m=0}^{M-1}\sum_{n=0}^{N-1}f(m,n)h(x-m,y-n)
    $$
    \item теорема о свертке
    $$
        \mathcal{F}[f\circ h]=\mathcal{F}[f]\cdot \mathcal{F}[h]
    $$
    \medskip
    $$
        \mathcal{F}[f\cdot h]=\mathcal{F}[f]\circ \mathcal{F}[h]
    $$
\end{itemize}
\end{frame}

%---------------------------------------------------------------------
\begin{frame}
\frametitle{Вычисление свертки с помощью преобразования Фурье}
\begin{figure}
    \centering
    \includegraphics[height=0.3\textheight]{fig9.pdf}
\end{figure}
\begin{itemize}
    \item  преобразование Фурье изображения и ядра свертки
    $$
        F=\mathcal{F}[f]\,,\qquad H=\mathcal{F}[h]
    $$
    \item теорема о свертке
    $$
        \mathcal{F}[f\circ h]=\mathcal{F}[f]\cdot \mathcal{F}[h]=F\cdot H
    $$
    \item вычисление свертки с помощью преобразования Фурье
    $$
        f\circ h =\mathcal{F}^{-1}[F\cdot H]
    $$
\end{itemize}
\end{frame}
